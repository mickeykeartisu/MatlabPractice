音声合成関数の既定値

                    ratioModeIndicator: 0

                        headRoomToClip: 22
               powerCheckSegmentLength: 15
                  timeAxisMappingTable: 1
      fundamentalFrequencyMappingTable: 1
             frequencyAxisMappingTable: 1
              timeAxisStretchingFactor: 1
                          DisplayPlots: 0
                              lowestF0: 50
                          statusReport: 'ok'
